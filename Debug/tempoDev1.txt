    '*****************************************************************
    ' controller changed notification - called
    ' after ALL HID devices are plugged or unplugged
    '*****************************************************************
    Public Sub OnChanged()
        ' get the handle of the device we are interested in, then set
        ' its read notify flag to true - this ensures you get a read
        ' notification message when there is some data to read...
        Dim pHandle As Integer
        pHandle = hidGetHandle(VendorID, ProductID)
        hidSetReadNotify(hidGetHandle(VendorID, ProductID), True)
    End Sub

    '*****************************************************************
    ' on read event...
    '*****************************************************************
    Public Sub OnRead(ByVal pHandle As Integer)
        ' read the data (don't forget, pass the whole array)...
        If hidRead(pHandle, BufferIn(0)) Then
            ' ** YOUR CODE HERE **
            ' first byte is the report ID, e.g. BufferIn(0)
            ' the other bytes are the data from the microcontroller...

            Timer3.Enabled = False                  ' Desable HID_Send for 50 mSec
            Timer3.Interval = 50
            Timer3.Enabled = True
            Select Case BufferIn(1)
                Case Is = &H8                       ' Response 0x08 GET_PART_INFO ( See AN649 )
                    Part_info()
                Case Is = &H9                       ' Response 0x09 GET_SYS_STATE ( See AN649 )
                    System_State()
                Case Is = &H12                      ' Response 0x12 GET_FUNC_INFO ( See AN649 )
                    Firmware_Revision()
                Case Is = &H80                      ' Response 0x80 GET_DIGITAL_SERVICE_LIST ( See AN649 )
                    Digital_Service_List()
                Case Is = &H84                      ' Response 0x84. GET_DIGITAL_SERVICE_DATA ( See AN649 )
                    Get_Digital_Service_Data()
                Case Is = &HB2                      ' Response 0xB2 DAB_DIGRAD_STATUS ( See AN649 )
                    DAB_Digrad_Status()
                Case Is = &HB4                      ' Response 0xB4 DAB_GET_ENSEMBLE_INFO ( See AN649 )
                    Ensemble_Info()
                Case Is = &HBC                      ' Response 0xBC DAB_GET_TIME ( See AN649 )
                    DAB_Get_Time()
                Case Is = &HBD                      ' Response 0xBD DAB_GET_AUDIO_INFO ( See AN649 )
                    DAB_Get_Audio_Info()
                Case Is = &HC0                      ' Response 0xC0 DAB_GET_SERVICE_INFO ( See AN649 )
                    DAB_Get_Service_Info()
                Case Is = &HF0                      ' Tuning for a new ensemble
                    Tuning_For_New_Ensemble()
                Case Is = &HFF                      ' Service List not available
                    Service_List_Not_Available()
            End Select
        End If
    End Sub

    Private Sub DAB_Digrad_Status()
        Tune_Frequency = ((BufferIn(20) * &H1000000) + (BufferIn(19) * &H10000) + (BufferIn(18) * &H100) + BufferIn(17)).ToString
        Tune_Index = BufferIn(21).ToString
        If BufferIn(11) < 128 Then
            RSSI = BufferIn(11) And &H3F    ' maxi 63
        Else
            RSSI = BufferIn(11) - 256
        End If
        Valid = BufferIn(10) And &H1
        SNR = BufferIn(12)
        CNR = BufferIn(14)
        FIC_Quality = BufferIn(13)
        Nb_Services = 0
        LabelRSSI.Text = RSSI.ToString & " dB�V"
        ProgressBarRSSI.Value = RSSI + 128
        If Command <> &H40 Then
            LabelSNR.Text = SNR.ToString & " dB"
            If SNR > 20 Then
                ProgressBarSNR.Value = 20
            Else
                ProgressBarSNR.Value = SNR
            End If
            LabelCNR.Text = CNR.ToString & " dB"
            If CNR > 54 Then
                ProgressBarCNR.Value = 54
            Else
                ProgressBarCNR.Value = CNR
            End If
            LabelFIC.Text = FIC_Quality.ToString
            If FIC_Quality > 100 Then
                ProgressBarFIC.Value = 100
            Else
                ProgressBarFIC.Value = FIC_Quality
            End If
        End If
        If Command = &H40 Then
            Form2.DataGridView1.Rows.Add(Tune_Index, Tune_Frequency, RSSI, Valid, Nb_Services)
            Row_Index = Row_Index + 1
            Form2.ProgressBar1.Value = Tune_Index

            If Form2.ProgressBar1.Value = 40 Then
                Form2.Label2.Text = "Scanning complete"
                Form2.ButtonCloseScan.Enabled = True

    Private Sub Soft_Labels_Artist_Title()
        Dim CharPosition As Integer
        Dim DLS_Len As Integer
        Dim SearchChar As String = " - "
        DLS_Len = Len(Dynamic_Label)
        CharPosition = InStr(1, Dynamic_Label, SearchChar, CompareMethod.Text)
        If CharPosition > 0 Then
            Title = Dynamic_Label.Substring(0, CharPosition - 1)
            Artist = Dynamic_Label.Substring(CharPosition + 2, DLS_Len - CharPosition - 2)
            TextBoxArtistDLS.Text = Artist
            TextBoxTitle.Text = Title
        Else
            TextBoxArtistDLS.Text = Dynamic_Label
        End If
    End Sub

    Private Sub Loading_Picture()
        If Picture_Hex <> "" Then
            Image_Start = False
            Path_Picture = "C:\SI4684\picture.jpg"
            Dim bytes As Byte() = New Byte(Picture_Hex.Length \ 2 - 1) {}
            For i As Integer = 0 To bytes.Length - 1
                bytes(i) = Convert.ToByte(Picture_Hex.Substring(i * 2, 2), 16)
            Next
            Dim fs As New FileStream(Path_Picture, FileMode.OpenOrCreate)
            fs.Write(bytes, 0, bytes.Length)
            fs.Close()
            Using my_stream As New IO.FileStream(Path_Picture, IO.FileMode.Open)

    Private Sub Load_Services()
        Dim fName As String = "C:\SI4684\services.txt"
        Dim TextLine As String = ""
        Dim SplitLine() As String
        If System.IO.File.Exists(fName) = True Then
            Using objReader As New System.IO.StreamReader(fName)
                Do While objReader.Peek() <> -1
                    TextLine = objReader.ReadLine()
                    SplitLine = Split(TextLine, ",")
                    Me.DataGridViewServices.Rows.Add(SplitLine)
                Loop
            End Using
        Else
            MessageBox.Show("Services list is empty : scan for services")
        End If
    End Sub
